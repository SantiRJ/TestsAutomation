package pageObjects;

import org.openqa.selenium.By;
import org.openqa.selenium.Keys;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.support.PageFactory;
import org.openqa.selenium.support.ui.ExpectedConditions;
import org.openqa.selenium.support.ui.WebDriverWait;

public class DespegarBusquedaPage {

	
	
	private static final WebElement AplicarPasajeros = null;
	private WebDriverWait wait;
	private WebDriver driver;
	
	

	public DespegarBusquedaPage(WebDriver driver) {
	this.driver=driver;
    PageFactory.initElements(driver,this);
    wait = new WebDriverWait (driver,5);
	}
	public boolean destinoEsVisible () {
	return this.destinoEsVisible(); 
	
	}


	public void ElegirDestino (String busqueda) throws InterruptedException{
	
			wait.until(ExpectedConditions.elementToBeClickable(destino)); 
			destino.click();
			Thread.sleep(2000);
			destino.sendkeys (busqueda);
			Thread.sleep(1000);
	        destino.sendkeys(Keys.ENTER);
	}
	


		public void elegirFecha () {

			By boxFecha;
			wait.until(ExpectedConditions.elementToBeClickable (boxFecha));
			((WebElement) boxFecha).click();
			By fechaSalida;
			wait.until(ExpectedConditions.elementToBeClickable(fechaSalida));
			fechaSalida.click();
			WebElement boxFechaRegreso;
			boxFechaRegreso.click();
			WebElement fechaRegreso;
			wait.until(ExpectedConditions.elementToBeClickable(fechaRegreso)); 
			fechaRegreso.click();
			WebElement aplicar;
			wait.until(ExpectedConditions.elementToBeClickable(aplicar));
			aplicar.click();
		}

		public void elegirPasajeros () {


			WebElement pasajeros = null;
			pasajeros.click();
			By mayores = null;
			wait.until(ExpectedConditions.elementToBeClickable (mayores));
			((WebElement) mayores).click();
			((WebElement) mayores).click();
			By menores = null;
			wait.until(ExpectedConditions.elementToBeClickable (menores));
			((WebElement) menores).click();
			By menuEdad = null;
			wait.until (ExpectedConditions.elementToBeClickable (menuEdad));
			((WebElement) menuEdad).click();
			WebElement edad = null;
			edad.click();
			AplicarPasajeros.click();

		}
}
